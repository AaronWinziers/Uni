def rechteck(p1, p2):

Polygone: Parts und Rings

# ring
rng = ((7.5, 4.3), (4.7, 5.4), (2.5, 2.1), (5.9, 1.1), (3.3,1.5))

def pr_closed(pr):
    # test auf geschlossenen polygonalen ring
    if pr[0][0] == pr[-1][0] and pr[0][1] == pr[-1][1]:
        return pr
    else:
        pr = list(pr)
        pr.append(pr[0])
        return pr

pol = pr_closed(rng)
print pol

Test und Korrektur der Drehrichtung: "signed area" nach Gauß

def pr_signed_area(pr):
        xs, ys = map(list, zip(*pr))
        xs.append(xs[1])
        ys.append(ys[1])
        return sum(xs[i]*(ys[i+1]-ys[i-1]) for i in range(1, len(pr)))/2.0

def pr_reverse(pr):
    temp = list(pr)
    temp.reverse()
    return tuple(temp)
